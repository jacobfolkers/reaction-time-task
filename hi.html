    <div id="circle"></div>


    <script>

        const WEB_APP_URL = "https://script.google.com/macros/s/AKfycbxiU03xHWGiwtL7HFd_BYW0eYtna3jeUVtjzyFHja_nD4CCy6_Fs7ZBb903xNUSlb76/exec";

        const TOTAL_TRIALS = 10;

        let participantId = "";

        let trialsCompleted = 0;

        let reactionTimes = [];

        let startTime = 0;


        const welcomeScreen = document.getElementById('welcome-screen');

        const resultsScreen = document.getElementById('results-screen');

        const circle = document.getElementById('circle');


        document.getElementById('start-btn').addEventListener('click', () => {

            participantId = document.getElementById('participant-id').value.trim() || "USER-" + Math.random().toString(36).substring(2, 8).toUpperCase();

            welcomeScreen.classList.add('hidden');

            startExperiment();

        });


        function startExperiment() {

            trialsCompleted = 0;

            reactionTimes = [];

            nextTrial();

        }


        function nextTrial() {

            if (trialsCompleted >= TOTAL_TRIALS) {

                submitAndCompare();

                return;

            }

            circle.style.display = 'none';

            setTimeout(() => {

                circle.style.left = `${Math.random() * (window.innerWidth - 100)}px`;

                circle.style.top = `${Math.random() * (window.innerHeight - 100)}px`;

                circle.style.display = 'block';

                startTime = performance.now();

            }, 500);

        }


        circle.addEventListener('mousedown', () => {

            reactionTimes.push(Math.round(performance.now() - startTime));

            trialsCompleted++;

            nextTrial();

        });


        async function submitAndCompare() {

            circle.classList.add('hidden');

            const userAverage = reactionTimes.reduce((a, b) => a + b, 0) / reactionTimes.length;

            

            document.getElementById('stats-summary').innerText = `Your Average: ${Math.round(userAverage)}ms. Fetching global data...`;

            resultsScreen.classList.remove('hidden');


            try {

                // 1. Submit current results

                await fetch(WEB_APP_URL, {

                    method: 'POST',

                    mode: 'no-cors',

                    body: JSON.stringify({ participantId, data: reactionTimes, average: userAverage })

                });


                // 2. GET historical results

                const response = await fetch(WEB_APP_URL);

                const historicalData = await response.json(); // Assumes Web App returns a flat array of all RTs

                

                renderChart(historicalData, userAverage);

                document.getElementById('stats-summary').innerText = `Your Average: ${Math.round(userAverage)}ms`;

            } catch (e) {

                console.error("Data sync failed", e);

                // Fallback: render chart with just user data if GET fails

                renderChart(reactionTimes, userAverage);

            }

        }


        function renderChart(allData, userAvg) {

            const ctx = document.getElementById('rtChart').getContext('2d');

            

            new Chart(ctx, {

                type: 'bar',

                data: {

                    labels: allData.map((_, i) => i), // Simple histogram bins

                    datasets: [{

                        label: 'All Participants RTs (ms)',

                        data: allData.sort((a, b) => a - b),

                        backgroundColor: 'rgba(0, 123, 255, 0.5)',

                    }]

                },

                options: {

                    responsive: true,

                    plugins: {

                        annotation: {

                            annotations: {

                                line1: {

                                    type: 'line',

                                    xMin: allData.findIndex(val => val >= userAvg), // Rough placement on sorted data

                                    xMax: allData.findIndex(val => val >= userAvg),

                                    borderColor: 'red',

                                    borderWidth: 4,

                                    label: {

                                        display: true,

                                        content: `You: ${Math.round(userAvg)}ms`,

                                        position: 'start'

                                    }

                                }

                            }

                        }

                    }

                }

            });

        }

    </script>
